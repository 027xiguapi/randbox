export const useGameStore = defineStore('rpsGame', {
  state: () => ({
    choices: ['rock', 'paper', 'scissors'],
    totalGames: 0,
    wins: 0,
    losses: 0,
    ties: 0,
    isLoading: false,
    results: []
  }),

      state.totalGames > 0 ? ((state.wins / state.totalGames) * 100).toFixed(1) : '0'

      else this.ties++
